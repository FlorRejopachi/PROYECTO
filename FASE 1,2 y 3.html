from sklearn.model_selection import train_test_split

X = df_encoded.drop('target', axis=1)

# Dividir el dataset

# Predecir
y_pred_nb = model_nb.predict(X_test)

# Evaluar
nb = accuracy_score(y_test, y_pred_nb)
